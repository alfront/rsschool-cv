/**
 * @param {number[]} nums
 * @return {number}
 */
function maxSubArray(nums) {
    let maxSum = -Infinity;
    let partSum = 0;

    for (let i = 0; i < nums.length; i++) {
        partSum = Math.max(nums[i], partSum + nums[i]);
        maxSum = Math.max(maxSum, partSum);
    }
    return maxSum;
};
